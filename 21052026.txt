SELECT MAX(salary) AS second_largest_salary
FROM worker
WHERE salary < (
    SELECT MAX(salary)
    FROM worker
    WHERE department = 'Admin'
)
AND department = 'Admin';

Select first_name, department, salary from worker where department='admin' order by salary desc limit 1 offset 1;

SELECT WORKER_ID,FIRST_NAME,DEPARTMENT,SALARY FROM Worker
WHERE DEPARTMENT='admin'
ORDER BY SALARY DESC
LIMIT 1 OFFSET 1;


SELECT department, first_name, salary
FROM (
    SELECT department,
           first_name,
           salary,
           DENSE_RANK() OVER (
               PARTITION BY department
               ORDER BY salary DESC
           ) AS rnk
    FROM worker
) AS temp
WHERE rnk = 2;

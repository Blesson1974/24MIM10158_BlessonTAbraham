create database office;

use office;
CREATE TABLE Departments (
dept_id INT PRIMARY KEY,
dept_name VARCHAR(50) NOT NULL,
location VARCHAR(50)
);

CREATE TABLE Jobs (
job_id INT PRIMARY KEY,
job_title VARCHAR(50),
min_salary DECIMAL(10,2),
max_salary DECIMAL(10,2)
);

INSERT INTO Jobs (job_id, job_title, min_salary, max_salary) VALUES

CREATE TABLE Employees (
emp_id INT PRIMARY KEY,
name VARCHAR(50) NOT NULL,
manager_id INT,
dept_id INT,
job_id INT,
salary DECIMAL(10,2),
hire_date DATE,
FOREIGN KEY (dept_id) REFERENCES Departments(dept_id),
FOREIGN KEY (job_id) REFERENCES Jobs(job_id)
);

-- 1.
SELECT 
    Employees.name AS Employee_Name,
    Jobs.job_title AS Job_Title,
    Departments.dept_name AS Department_Name
FROM Employees
INNER JOIN Jobs 
    ON Employees.job_id = Jobs.job_id
INNER JOIN Departments 
    ON Employees.dept_id = Departments.dept_id;

 -- 2.
SELECT 
    name,
    salary
FROM Employees
WHERE salary >
(
    SELECT AVG(salary)
    FROM Employees
    WHERE dept_id =
    (
        SELECT dept_id
        FROM Departments
        WHERE dept_name = 'IT'
    )
);

-- 3.
SELECT 
    Departments.dept_name AS Department_Name,
    SUM(Employees.salary) AS Total_Salary
FROM Employees
INNER JOIN Departments
    ON Employees.dept_id = Departments.dept_id
GROUP BY Departments.dept_name;


-- 5.
SELECT 
    Employee.name AS Employee_Name,
    Manager.name AS Manager_Name
FROM Employees AS Employee
LEFT JOIN Employees AS Manager
    ON Employee.manager_id = Manager.emp_id;


-- 6.
SELECT 
    name,
    salary
FROM Employees
ORDER BY salary DESC
LIMIT 3;

-- 7.
SELECT 
    Employees.name AS Employee_Name
FROM Employees
INNER JOIN Jobs
    ON Employees.job_id = Jobs.job_id
INNER JOIN Departments
    ON Employees.dept_id = Departments.dept_id
WHERE Jobs.job_title = 'Developer'
  AND Departments.location = 'San Francisco';
  
  
  -- 8.
  SELECT 
    YEAR(hire_date) AS Hire_Year,
    COUNT(emp_id) AS Total_Employees
FROM Employees
GROUP BY YEAR(hire_date);

-- 10.
SELECT 
    name
FROM Employees
WHERE dept_id =
(
    SELECT dept_id
    FROM Employees
    WHERE name = 'David'
);

-- 11.
SELECT 
    Departments.dept_name AS Department_Name,
    Employees.name AS Employee_Name
FROM Departments
LEFT JOIN Employees
    ON Departments.dept_id = Employees.dept_id;
    
-- 12.
SELECT 
    name,
    salary,
    hire_date
FROM Employees
WHERE hire_date BETWEEN '2018-01-01' AND '2020-12-31'
  AND salary > 60000;
  
  
-- 13.
SELECT 
    manager_id AS Manager_ID,
    MIN(salary) AS Minimum_Salary
FROM Employees
WHERE manager_id IS NOT NULL
GROUP BY manager_id;

-- 16
SELECT 
    dept_name
FROM Departments
WHERE dept_id NOT IN
(
    SELECT DISTINCT Employees.dept_id
    FROM Employees
    INNER JOIN Jobs
        ON Employees.job_id = Jobs.job_id
    WHERE Jobs.job_title = 'Analyst'
);

-- 18
SELECT 
    Employees.name AS Employee_Name,
    Employees.hire_date AS Hire_Date,
    Departments.dept_name AS Department_Name,
    Employees.salary
FROM Employees
INNER JOIN Departments
    ON Employees.dept_id = Departments.dept_id
ORDER BY Departments.dept_name ASC,
         Employees.salary DESC;
         
-- 20
SELECT 
    name,
    salary
FROM Employees
WHERE salary =
(
    SELECT DISTINCT salary
    FROM Employees
    ORDER BY salary DESC
    LIMIT 1 OFFSET 2
);

-- 21 
SELECT 
    Jobs.job_title AS Job_Title,
    COUNT(Employees.emp_id) AS Employee_Count
FROM Jobs
LEFT JOIN Employees
    ON Jobs.job_id = Employees.job_id
GROUP BY Jobs.job_title;

-- 22.
SELECT 
    Employees.name AS Employee_Name
FROM Employees
INNER JOIN Departments
    ON Employees.dept_id = Departments.dept_id
WHERE Departments.dept_name = 'Finance'
  AND Employees.name LIKE '%i%';
  
  
-- 24
SELECT 
    dept_name
FROM Departments
WHERE dept_id =
(
    SELECT dept_id
    FROM Employees
    GROUP BY dept_id
    ORDER BY SUM(salary) DESC
    LIMIT 1
);
